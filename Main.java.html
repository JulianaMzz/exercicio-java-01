                maior2 = num;

        }

    public static void ex20() {
        Scanner scanner = new Scanner(System.in);

        // Entrada de dados
        System.out.print("Nome do produto: ");
        String nomeProduto = scanner.nextLine();

        System.out.print("Quantidade adquirida: ");
        int quantidade = scanner.nextInt();

        System.out.print("Preço unitário (R$): ");
        double precoUnitario = scanner.nextDouble();

        // Cálculo do valor da venda
        double valorVenda = quantidade * precoUnitario;

        // Determinação do percentual de desconto
        double percentualDesconto;
        if (quantidade <= 5) {
            percentualDesconto = 0.02;  // 2%
        } else if (quantidade <= 10) {
            percentualDesconto = 0.03;  // 3%
        } else {
            percentualDesconto = 0.05;  // 5%
        }

        // Cálculo do desconto e imposto
        double desconto = valorVenda * percentualDesconto;
        double imposto = valorVenda * 0.20;

        double totalPagar = valorVenda - desconto + imposto;

        System.out.println("\n======= NOTA FISCAL =======");
        System.out.println("Produto: " + nomeProduto);
        System.out.println("Quantidade: " + quantidade);
        System.out.printf("Preço unitário: R$ %.2f\n", precoUnitario);
        System.out.printf("Valor da venda: R$ %.2f\n", valorVenda);
        System.out.printf("Percentual de desconto: %.0f%%\n", percentualDesconto * 100);
        System.out.printf("Valor do desconto: R$ %.2f\n", desconto);
        System.out.printf("Imposto (20%%): R$ %.2f\n", imposto);
        System.out.printf("Total a pagar: R$ %.2f\n", totalPagar);
        System.out.println("==========================");

        scanner.close();

        ex21();
    }

        if (horasTrabalhadas <= horasNormaisMes) {

            int horasExtras = horasTrabalhadas - horasNormaisMes;

    public static void ex25() {
        Scanner scanner = new Scanner(System.in);
        String[] array = new String[12];

        System.out.println("Informe 12 valores (letras ou números):");
        for (int i = 0; i < 12; i++) {
            array[i] = scanner.next();
        }

        System.out.print("Saída 1: ");
        for (int i = 11; i >= 0; i--) {
            System.out.print(array[i]);
            if (i > 0) System.out.print(", ");
        }
        System.out.println();

        int countNumeros = 0;
        int countLetras = 0;
        for (String valor : array) {
            if (valor.matches("\\d+")) {
                countNumeros++;
            } else {
                countLetras++;
            }
        }

        String[] numeros = new String[countNumeros];
        String[] letras = new String[countLetras];


        int idxNum = 0, idxLet = 0;
        for (String valor : array) {
            if (valor.matches("\\d+")) {
                numeros[idxNum++] = valor;
            } else {
                letras[idxLet++] = valor;
            }
        }

        System.out.print("Saída 2: ");
        for (int i = numeros.length - 1; i >= 0; i--) {
            System.out.print(numeros[i]);
            if (i > 0) System.out.print(", ");
        }

        System.out.print(" - ");

        for (int i = letras.length - 1; i >= 0; i--) {
            System.out.print(letras[i]);
            if (i > 0) System.out.print(", ");
        }
        System.out.println();

        scanner.close();
    }
